Attribute VB_Name = "VBScript"
Sub Format_Excel()
'To loop the worksheet/workbook to warp text, sws stand for starting worksheet, ws stand for worksheet'
Dim ws As Worksheet
Dim sws As Worksheet
Set sws = ActiveSheet 'remember which worksheet is active from the start'

For Each ws In ThisWorkbook.Worksheets
ws.Activate

'if you want to unwrap the text change it to false, the rest is all border and '
Cells.WrapText = True
Cells.ColumnWidth = 25
Cells.Borders.LineStyle = xlContinous
Cells.Borders.Weight = xlThin

'to freeze top rows to off it just change the true to false'

With ActiveWindow
    .SplitColumn = 0
    .SplitRow = 1
End With
ActiveWindow.FreezePanes = True

'to color/highlight the selected cells pls google the color you want example blue rgb '

Range("A1").EntireColumn.Interior.Color = RGB(255, 255, 0)
Range("H1").EntireColumn.Interior.Color = RGB(0, 150, 255)
Range("L1").EntireColumn.Interior.Color = RGB(0, 150, 255)
Range("J1:K1").EntireColumn.Interior.Color = RGB(255, 255, 0)
Range("M1:N1").EntireColumn.Interior.Color = RGB(255, 255, 0)

'to bold the entire column to off the bold just change it to false'

Range("A1").EntireColumn.Font.Bold = True
Range("H1").EntireColumn.Font.Bold = True
Range("L1").EntireColumn.Font.Bold = True
Range("J1:K1").EntireColumn.Font.Bold = True
Range("M1:N1").EntireColumn.Font.Bold = True

Next

sws.Activate 'activate the worksheet that was first started from the workbook'

End Sub
